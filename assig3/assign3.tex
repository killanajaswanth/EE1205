 At, t=0^-

after t=0,\\
KVL, \begin{align}
2i\brak{t}+L\frac{di}{dt}
\end{align}
apply laplace transform,
 \begin{align}
2I\brak{s}-Li\brak{{0^{-}}}+LsI\brak{s}&=0\\
\implies I\brak{s}&=\frac{i\brak{{0^{-}}}}{s+2}\\
I\brak{s}&=\frac{2}{s+2}
 \end{align}
 applying inverse laplace transform
 \begin{align}
  I\brak{t}&=2\cdot e^{-2t}\\
  V_R&=-2I\brak{t}\\
  \implies V_R&=-4\cdot e^{-2t}
 \end{align}  
  As,
 \begin{align}
    t & \xrightarrow{} 0\\
     e^{-2t}&\xrightarrow{} 1\\
     \abs{V_R\brak{max}}&=4
 \end{align}
